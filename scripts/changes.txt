void c_gui::Settting()
{
    auto size = ImGui::GetCurrentWindow()->Size;
    ImGui::SetCursorPos(ImVec2(10, 20));
    ImGui::BeginChild(obf("SettingArea"), ImVec2(size.x - 20, size.y - 30), false);

    ImGui::SeparatorText(obf(u8"Current Mode"));
    ImGui::Combo(obf(u8"Mode"), &set->Setting.game_mode, u8"All-Out Warfare\0Hazard Zone\0");

    ImGui::SeparatorText(obf(u8"Hotkeys"));
    Hotkey(obf(u8"Show/Hide Menu"), &set->Setting.MemuKey);
    Hotkey(obf(u8"Battle Mode (F2)"), &set->Setting.BattleMode);

    ImGui::Text(set->Setting.战斗模式 ? u8"On - Hide Loot" : u8"Off");

    ImGui::SeparatorText(obf(u8"Configuration"));
    if (ImGui::Button(obf(u8"Load Config"))) set->load();
    if (ImGui::Button(obf(u8"Save Config"))) set->save();

    ImGui::SeparatorText(obf(u8"Other"));
    if (ImGui::Button(set->Setting.ISBattleMode ? u8"Disable Anti-Screenshot" : u8"Enable Anti-Screenshot"))
    {
        HWND hwnd = Overlay::GetHWND();
        if (hwnd)
        {
            set->Setting.ISBattleMode = !set->Setting.ISBattleMode;
            SetWindowDisplayAffinity(hwnd,
                set->Setting.ISBattleMode ? WDA_EXCLUDEFROMCAPTURE : WDA_NONE);
        }
    }

    ImGui::SeparatorText(obf(u8"Cursor Settings"));

    if (ImGui::Button(showCursor ? u8"Disable Cursor" : u8"Enable Cursor"))
    {
        showCursor = !showCursor;

        if (showCursor)
        {
            while (ShowCursor(TRUE) < 0);
        }
        else
        {
            while (ShowCursor(FALSE) >= 0);
        }
    }

    Hotkey(obf(u8"Cursor Hotkey (F1)"), &set->Setting.cursorkey);

    ImGui::EndChild();
}
