    bool updateseed_todriver(ULONG64 biosseed, ULONG64 diskseed)
    {
        REQUEST_DATA req{};
        req.SecurityCode1 = SECURITY_CODE1;
        req.SecurityCode2 = SECURITY_CODE2;
        req.Command = RequestId::Set_Spoof_Seed;
        req.SpoofData.BIOS_SEED = biosseed;
        req.SpoofData.DISK_SEED = diskseed;

        if (SendIoctl(&req) && req.Result)
        {
            return true;
        }
        return false;
    }
